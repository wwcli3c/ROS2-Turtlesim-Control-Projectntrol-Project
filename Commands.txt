# Start the turtlesim window
ros2 run turtlesim turtlesim_node

# Open another terminal and control the turtle with keyboard
ros2 run turtlesim turtle_teleop_key

# Check all available topics
ros2 topic list

# Move turtle1 by publishing to the cmd_vel topic
ros2 topic pub /turtle1/cmd_vel geometry_msgs/msg/Twist "{linear: {x: 2.0}, angular: {z: 1.8}}"

# Teleport turtle1 to a specific position (x, y, theta)
ros2 service call /turtle1/teleport_absolute turtlesim/srv/TeleportAbsolute "{x: 5.0, y: 5.0, theta: 1.57}"

# Change background color to black
ros2 param set /turtlesim background_r 0
ros2 param set /turtlesim background_g 0
ros2 param set /turtlesim background_b 0

# Clear the screen to apply background color
ros2 service call /clear std_srvs/srv/Empty

# Spawn a second turtle called 'turtle2'
ros2 service call /spawn turtlesim/srv/Spawn "{x: 2.0, y: 2.0, theta: 0.0, name: 'turtle2'}"

# Move turtle2 using its own cmd_vel topic
ros2 topic pub /turtle2/cmd_vel geometry_msgs/msg/Twist "{linear: {x: 1.5}, angular: {z: 0.5}}"

# Remove (kill) turtle2 from the simulation
ros2 service call /kill turtlesim/srv/Kill "{name: 'turtle2'}"
